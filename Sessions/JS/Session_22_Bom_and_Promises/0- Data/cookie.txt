// 1. `path`
// يحدد النطاق الذي ستكون فيه الكوكيز متاحة.
// مثال: جعل الكوكيز متاحة لجميع صفحات الموقع
document.cookie = "name=ahmed; path=/";

// 2. `expires`
// يحدد وقت انتهاء صلاحية الكوكيز. يجب أن يكون بصيغة تاريخ UTC.
// مثال: انتهاء صلاحية الكوكيز في 31 ديسمبر 2024
document.cookie = "name=ahmed; expires=Fri, 31 Dec 2024 23:59:59 GMT; path=/";

// 3. `max-age`
// يحدد فترة صلاحية الكوكيز بالثواني من لحظة تعيينها. بديل لـ `expires`.
// مثال: الكوكيز صالحة لمدة ساعة (3600 ثانية)
document.cookie = "name=ahmed; max-age=3600; path=/";

// 4. `domain`
// يحدد النطاق الذي تكون فيه الكوكيز متاحة. يمكنك توسيع النطاق إلى نطاقات فرعية.
// مثال: الكوكيز ستكون متاحة لـ `example.com` و `sub.example.com`
document.cookie = "name=ahmed; domain=example.com; path=/";

// 5. `secure`
// يشير إلى أن الكوكيز يجب أن تُرسل عبر اتصالات HTTPS فقط.
// مثال: جعل الكوكيز متاحة فقط عبر HTTPS
document.cookie = "name=ahmed; secure; path=/";

// 6. `HttpOnly`
// يحدد أن الكوكيز لا يمكن الوصول إليها عبر JavaScript. يحسن الأمان ضد هجمات XSS.
// مثال: منع الوصول إلى الكوكيز عبر JavaScript
document.cookie = "name=ahmed; HttpOnly; path=/";

// 7. `SameSite`
// يحدد كيفية إرسال الكوكيز مع الطلبات عبر المواقع. القيم الممكنة: `Strict`, `Lax`, `None`.
// `Strict`: يمنع إرسال الكوكيز مع الطلبات من مواقع أخرى.
// `Lax`: يسمح بإرسال الكوكيز مع بعض الطلبات عبر المواقع (مثل الروابط).
// `None`: يسمح بإرسال الكوكيز مع جميع الطلبات عبر المواقع (يجب استخدام `Secure` مع `None`).
// مثال: إرسال الكوكيز مع الطلبات عبر المواقع في وضع `Lax`
document.cookie = "name=ahmed; SameSite=Lax; path=/";
