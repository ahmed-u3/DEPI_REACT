cd /folder name
=> علشان نغير المسار بتاعنا ونتحرك من فولدر للتاني
ex: cd /desktop

cd ..
=> علشان نرجع خطوة لورا

dir 
=> هيعملك ليست بالفولدرات والفايلات في الفولدر اللي انت واقف عليه جوه ال cmd

mkdir "new folder"
=> علشان نعمل فولدر

rmdir test
=> هيمسح الفولدر لو فاضي

rmdir /s /q
=> هيمسح الفولدر بكل اللي فيه حتي الفايلات

echo > file.txt
=> هيعمل فايل جديد

echo some text > file.txt
=> هيعمل فايل جديد جواه جملة some text

del file.txt
هيمسح فايل file.txt

del /s /q folder
هيمسح كل الفايلات من الفولدر بس الفولدرات نفسها هتفضل


ren folder "new folder"
هيغير اسم الفولدر
